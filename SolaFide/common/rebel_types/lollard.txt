######################################################################
# Lollard Rebels. Sola Fide: Modified to work with both new Lollard religion and
# the existing lollard events
######################################################################
lollard_rebels = {

	color = { 235 115 115 }

	area = nation
	government = theocracy 			# Desires Theocracy
	religion = lollard
	defection = none					# Will defect to the "best" alternative.
	defect_delay = 120
	independence = none				# Does not strive for independence
	unit_transfer = no
	gfx_type = culture_province	
	will_relocate = yes	
	
	resilient = no
	reinforcing = yes
	general = yes
	smart = yes
	
	artillery = 0.0
	infantry = 0.7
	cavalry = 0.3
	
	morale = 1.0
	
	# Possible handle actions
	handle_action_negotiate = yes
	handle_action_stability = yes
	handle_action_build_core = yes
	handle_action_send_missionary = yes		
	
	# The rebel type with the highest modifier for this province gets picked
	spawn_chance = {
		factor = 1
		modifier = {
			factor = 0.5 # until the Lollard Heresy event fires, make this much less likely
			NOT = { has_global_flag = lollard_heresy }
		}
		# TODO: is this modifier necessary with the minority mechanic?
		modifier = {
			factor = 1000
			has_province_modifier = sola_fide.lollards
		}
		modifier = {
			factor = 0
			NOT = { owner = { religion = ROOT } }
		}
		modifier = {
			factor = 0
			NOT = { owner = { religion = catholic } }
		}
		modifier = {
			factor = 0
			religion_group = pagan
		}
		modifier = {
			factor = 0
			religion_group = eastern
		}
		modifier = {
			factor = 0.5
			is_year = 1700
		}
		modifier = {
			factor = 0.01
			is_overseas = yes
		}
		modifier = {
			factor = 5
			has_province_modifier = "heresy"
		}
		modifier = {
			factor = 3.0
			is_year = 1450
			owner = { religion = catholic }
			NOT = { is_religion_enabled = protestant }
			NOT = { is_religion_enabled = reformed }
		}
		modifier = {
			factor = 0.0
			NOT = {
				religion = lollard
				has_province_modifier = "sola_fide.religious_minority.lollard"
			}
		}
		modifier = {
			factor = 10
			has_province_modifier = "sola_fide.religious_minority.lollard"
		}
	}
 
 	# This is checked for EACH province in the Area of Operations
	movement_evaluation = {
		factor = 1
		modifier = {
			factor = 0.001
			has_province_modifier = "heresy"
		}
		modifier = {
			factor = 0.1
			units_in_province = 1
		}
		modifier = {
			factor = 0.001
			controlled_by = REB
		}
		modifier = {
			factor = 1.5
			unrest = 2
		}
		modifier = {
			factor = 1.5
			unrest = 4
		}
		modifier = {
			factor = 1.5
			unrest = 6
		}
		modifier = {
			factor = 2
			NOT = { religion = lollard }
		}
		modifier = {
			factor = 10
			AND = {
				NOT = { religion = lollard }
				has_province_modifier = "sola_fide.religious_minority.lollard"
			}
		}
	}
 
 	# Province Scope
	siege_won_effect = {
		custom_tooltip = religious_rebels_seige_effect_tooltip
		hidden_effect = {
			if = {
				limit = {
					OR = {
						religion = lollard
						is_reformation_center = yes
						owner = { religion = lollard }
					}
				}
				add_province_modifier = {
					name = "religious_violence"
					duration = 730
				}
			}
			if = {
				limit = {
					NOT = { religion = lollard }
					is_reformation_center = no
					NOT = { owner = { religion = lollard } }
				}
				change_religion = lollard
				add_province_modifier = {
					name = "forced_conversion"
					duration = 730
				}
			}
		}
		owner = {
			change_estate_land_share = {
				estate = estate_church
				share = 10.0
				province = ROOT
			}
		}
	}
	
	# Country scope
	can_negotiate_trigger = {
		NOT = { tag = PAP }
	}
	
	# Country scope
	can_enforce_trigger = {
		NOT = { tag = PAP }
	}
	
	# Localisation for their demands
	demands_description = "lollard_rebels_demand"
	
	# Country Scope	
	demands_enforced_effect = {
		hidden_effect = {
			add_country_modifier = {
				name = just_lost_to_rebels
				duration = 10
				hidden = yes
			}
		}
		add_country_modifier = {
			name = eng_tolerates_lollards
			duration = 36500
		}
		remove_country_modifier = eng_opposes_lollards
		add_papal_influence = -100
		add_prestige = -50
		every_owned_province = {
			remove_province_modifier = sola_fide.lollards
		}

		# Add effects that would normally apply if you lose to religous rebels

		if = {
			limit = {
				NOT = { religion_group = christian }
			}
			add_prestige = -50
			every_owned_province = {
				limit = {
					has_rebel_faction = REB
				}
				add_local_autonomy = 20
			}
		}
		if = {
			limit = {
				tag = PAP
			}
			add_stability = -1
		}
		else_if = {
			limit = {
				religion = lollard
			}
			add_country_modifier = {
				name = "religious_intolerance_enforced"
				duration = 3650
			}
		}
		else_if = {
			limit = {
				NOT = { religion = lollard }
				dominant_religion = lollard
			}
			change_religion = lollard
			force_converted = yes
		}
		else_if = {
			limit = {
				NOT = { religion = lollard }
				NOT = { dominant_religion = lollard }
				religion_group = christian
				NOT = { has_country_modifier = heretic_tolerance }
			}
			
			add_country_modifier = {
				name = "heretic_tolerance"
				duration = 3650
			}
		}
		else = { add_stability = -1 } #Should probably never get here
		if = {
			limit = {
				religion = lollard
				has_estate = estate_church
				NOT = {
					has_estate_influence_modifier = {
						estate = estate_church
						modifier = EST_VAL_RELIGIOUS_ZEALOTS_TRIUMPH
					}
				}
			}
			add_estate_influence_modifier = {
				estate = estate_church
				desc = EST_VAL_RELIGIOUS_ZEALOTS_TRIUMPH
				duration = 3650
				influence = 15
			}
		}

		change_estate_land_share = {
			estate = estate_church
			share = 10.0
		}
	}
}
